# Par-ou-mpar---prova

#include <iostream>
using namespace std;


int main()
{
    int num1, num2;
    
    cout<<"Insira um número: "<<endl;
    cin>>num1;
    
    if(num1%2==1){
        cout<<"Este número é ímpar."<<endl;
    }
    else{
        cout<<"Este número é par."<<endl;
    }
    cout<<"Insira um segundo número: "<<endl;
    cin>>num2;
    
    if(num2%2==1){
        cout<<"Este número é ímpar.";
    }
    else{
        cout<<"Este número é par.";
        
    }


    return 0;
}
